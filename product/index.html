Product Detail Page

Change Material : 

 await v3.[V3 Viewer Name]
     .changeMaterial( [part name], [Color Name]); 
 await v3.main.changeMaterial('UPHLOSTREY', 'LINENDENIM');

Change Camera View : 

 await v3.[V3 Viewer Name]
     .setCameraView(  { angel_x , angel_y, distance ,  ... } ); 
 await v3.main.setCameraView({ angle_x:0  ,angle_y:  90   });

Change Camera zoom (fov) : 

 await v3.main.setCameraView({ zoom:  10   });

Group Material Change : 

 await v3.[V3 Viewer Name]
     .groupMaterial({
         GROUP1 : [MATERIAL SETTING], ...},
         callback); 
 await v3.main.groupMaterial( {
      BUTTONS : {code:'BUTTONMAT',color:'#AA8800'}
    });

Group Visibility : 

 await v3.[V3 Viewer Name]
     .groupVisibility("/** show group list A,B,.. ** /","/* hide group list   A,B,.. */",callback); 
 await v3.main.groupVisibility("BTN2","BUTTONS");

Switch Viewer : 

  <v3-image
     name="View Name"
     viewer="Main Viewer Name" >
 </v3-image>
                        
  v3.[V3 Viewer Name].switchToViewer( [view name] );
  v3.[V3 Viewer Name].switchToMain(); 
 v3.main.switchToViewer('VIEW1');

Request Realistic Image : 

  
                        
  v3.requestRealisticImage(v3.[V3 Viewer Name],callback);  
 v3.requestRealisticImage(v3.main,
      document.getElementById('promp').value,
      function(url){document.getElementById('IMAGE_VIEW1').src = url;});
